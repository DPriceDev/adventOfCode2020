cmake_minimum_required(VERSION 3.17)

target_link_libraries(day07_part1 apollo)
target_include_directories(day07_part1 PUBLIC .)

add_custom_command(
        TARGET day07_part1 POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy
        ${CMAKE_CURRENT_SOURCE_DIR}/input.txt
